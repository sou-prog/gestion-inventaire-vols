# Use a base image with Java 17 (or the version you need)
FROM openjdk:17-jdk-slim

# Set the working directory inside the container
WORKDIR /app

# Copy the Spring Boot jar file from your local machine to the container
COPY target/myproject-0.0.1-SNAPSHOT.jar app.jar

# Expose the port the app will run on (usually 8080)
EXPOSE 8080

# Command to run your app
ENTRYPOINT ["java", "-jar", "app.jar"]
